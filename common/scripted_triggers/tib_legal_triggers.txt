		government_has_flag = government_is_islamic
		custom_description = {
			text = "has_law_allowing_designate_heir"
			has_realm_law_flag = can_designate_heirs
		}
	}

		text = "dynasty_member_leads_election"
		NAND = {

		}
	}	
}
