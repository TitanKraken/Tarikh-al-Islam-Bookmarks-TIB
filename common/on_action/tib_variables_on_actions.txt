on_game_start = {
	on_actions = {
		on_islamic_start
	}
}
yearly_global_pulse = {
	on_actions = {
		tib_islamic_schism
	}
}
tib_islamic_schism = {
	effect = {
		if = {
			limit = {
				# Game started in 1066 or earlier
				game_start_date >= 950.1.1
				current_date >= 950.1.1
			}
			religion = {
				every_faith = {
					limit = {
						faith = faith:ashari # Sunni
					}
					every_faith_character = {
						trigger_event = tib_islamic_schism.0001
					}
				}
				every_faith = {
					limit = {
						faith = faith:ismaili # Shia
					}
					every_faith_character = {
						trigger_event = tib_islamic_schism.0010
					}
				}
			}
		}
	}
}

on_islamic_start = {
	effect = {
		every_ruler = {
			limit = { 
				#NOT = {exists = var:cal_authority}
				has_government = islamic_government
			}
			#save_scope_as = caliph
			#set_variable = {
			#	name = cal_authority
			#	value = 10
			#}
			set_variable = {
				name = zakat_resource
				value = 0
			}
        }
    }
}
